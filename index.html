const video = document.getElementById('video'); const canvas = document.getElementById('canvas'); const ctx = canvas.getContext('2d', { willReadFrequently: true }); const statusDiv = document.getElementById('status'); const debugDiv = document.getElementById('debug'); // 確保這是您的專屬 Google API const API_URL = "https://script.google.com/macros/s/AKfycbyBVSH5XNEpiaPWhDoD4AhEURGxxhKSlTbqlpt4gg4CvQkkPC4qLn2WMwqFWkUb9niePw/exec"; const COOLDOWN_MS = 60000; // 【新增】：連續命中計數器 let validFrameCount = 0; function startCamera() { navigator.mediaDevices.getUserMedia({ video: { facingMode: "environment", frameRate: { ideal: 30, max: 60 } } }) .then(function(stream) { video.srcObject = stream; statusDiv.innerText = "請用「紅色掃描線」對準發射器..."; statusDiv.style.color = "#007bff"; setInterval(analyzeStripes, 50); }) .catch(function(err) { statusDiv.innerText = "無法啟動相機：" + err; }); } function analyzeStripes() { if (video.readyState === video.HAVE_ENOUGH_DATA) { canvas.width = video.videoWidth; canvas.height = video.videoHeight; ctx.drawImage(video, 0, 0, canvas.width, canvas.height); const centerX = canvas.width / 2; const centerY = canvas.height / 2; const startY = centerY - 70; const scanHeight = 140; const imageData = ctx.getImageData(centerX, startY, 1, scanHeight); const data = imageData.data; let brightnessValues = []; let totalBrightness = 0; for (let i = 0; i < data.length; i += 4) { let brightness = (data[i] + data[i+1] + data[i+2]) / 3; brightnessValues.push(brightness); totalBrightness += brightness; } let averageBrightness = totalBrightness / brightnessValues.length; let maxB = brightnessValues[0]; let minB = brightnessValues[0]; for(let b of brightnessValues) { if(b > maxB) maxB = b; if(b < minB) minB = b; } let contrast = maxB - minB; let crossings = 0; let lastSign = (brightnessValues[0] >= averageBrightness) ? 1 : -1; let lastCrossingIndex = 0; let stripeWidths = []; for (let i = 1; i < brightnessValues.length; i++) { let currentSign = (brightnessValues[i] >= averageBrightness) ? 1 : -1; if (currentSign !== lastSign) { crossings++; stripeWidths.push(i - lastCrossingIndex); lastCrossingIndex = i; lastSign = currentSign; } } let isRegular = false; if (stripeWidths.length > 3) { let validWidths = stripeWidths.slice(1, -1); let minW = Math.min(...validWidths); let maxW = Math.max(...validWidths); if ((maxW - minW) <= 10) { isRegular = true; } } debugDiv.innerText = `亮度:${Math.round(averageBrightness)} | 對比:${Math.round(contrast)} | 條紋:${crossings} | 規律:${isRegular}`; // ========================================== // 嚴密過濾邏輯 + 連續影格驗證 // ========================================== let isSignalValid = false; // 條件一：對比度大於 200 (徹底封殺環境光) // 條件二：條紋大於等於 3 條 // 條件三：必須具有規律性 (isRegular) if (contrast > 200 && averageBrightness > 80 && crossings >= 3 && isRegular) { isSignalValid = true; } // 如果正在傳送中，就不要干擾畫面 if (isSending) return; if (isSignalValid) { validFrameCount++; // 命中一次，加一分 if (validFrameCount < 3) { statusDiv.innerText = `🔍 捕捉到特徵，驗證中... (${validFrameCount}/3)`; statusDiv.style.color = "orange"; } else { // 連續 3 次完美命中，才是真正的發射器！ sendData("A"); validFrameCount = 0; // 重置計數器 } } else { // 只要有一幀不符合（例如手抖移開，或是掃到環境雜訊），計數器瞬間歸零！ validFrameCount = 0; // 保持顯示預設文字 (避免洗掉防刷警告) if (statusDiv.innerText.includes("驗證中")) { statusDiv.innerText = "系統就緒，等待掃描..."; statusDiv.style.color = "#2c3e50"; } } } } let isSending = false; function sendData(roomCode) { const studentId = document.getElementById('studentId').value.trim(); if(!studentId) { statusDiv.innerText = "❌ 驗證通過，但您忘記輸入學號！"; statusDiv.style.color = "red"; return; } const now = Date.now(); const storageKey = 'lastCheckIn_' + studentId; const lastCheckIn = localStorage.getItem(storageKey); if (lastCheckIn) { const timePassed = now - parseInt(lastCheckIn); if (timePassed < COOLDOWN_MS) { let remainSeconds = Math.ceil((COOLDOWN_MS - timePassed) / 1000); statusDiv.innerText = `⏳ 學號 ${studentId} 請勿狂刷！請等待 ${remainSeconds} 秒。`; statusDiv.style.color = "red"; return; } } isSending = true; statusDiv.innerText = "🔒 連續驗證通過！正在傳送至雲端..."; statusDiv.style.color = "#007bff"; const fetchUrl = `${API_URL}?student_id=${studentId}&room=${roomCode}`; fetch(fetchUrl) .then(response => response.text()) .then(result => { localStorage.setItem(storageKey, Date.now()); statusDiv.innerText = "🎉 " + result + " (打卡成功)"; statusDiv.style.color = "green"; setTimeout(() => { isSending = false; statusDiv.innerText = "系統就緒，等待掃描..."; statusDiv.style.color = "#2c3e50"; }, 3000); }) .catch(error => { statusDiv.innerText = "網路連線失敗，請檢查網路"; isSending = false; }); }
